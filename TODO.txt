void	update_status(t_shell *data)
{
	if (WIFEXITED(data->status))
		data->status = WEXITSTATUS(data->status);
	else if (WIFSIGNALED(data->status)
		&& WTERMSIG(data->status) != SIGHUP)
		data->status = 128 + data->status;
}
